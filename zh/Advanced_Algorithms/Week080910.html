def factorial_recur(n):
    if n == 0:
        return 1
    count = 0
    for i in range(n):
        count += factorial_recur(n - 1)
        
    return count

def calculate_distance(city1, city2):
    """
    è®¡ç®—åŸå¸‚ city1 å’Œ city2 ä¹‹é—´çš„æ¬§å‡ é‡Œå¾—è·ç¦»ã€‚
    :param city1: åŸå¸‚1çš„åæ ‡ (x1, y1)
    :param city2: åŸå¸‚2çš„åæ ‡ (x2, y2)
    :return: ä¸¤ä¸ªåŸå¸‚ä¹‹é—´çš„è·ç¦»
    """
    return math.sqrt((city1[0] - city2[0]) ** 2 + (city1[1] - city2[1]) ** 2)

def total_distance(path, cities):
    """
    è®¡ç®—ç»™å®šè·¯å¾„çš„æ€»è·ç¦»ã€‚
    :param path: åŸå¸‚è®¿é—®é¡ºåºçš„åˆ—è¡¨
    :param cities: åŸå¸‚çš„åæ ‡åˆ—è¡¨
    :return: è·¯å¾„çš„æ€»è·ç¦»
    """
    distance = 0
    for i in range(len(path)):
        city1 = cities[path[i]]
        city2 = cities[path[(i + 1) % len(path)]]
        distance += calculate_distance(city1, city2)
    return distance

def greedy_construct_solution(cities):
    """
    ä½¿ç”¨éšæœºåŒ–è´ªå¿ƒç®—æ³•æ„é€ ä¸€ä¸ªåˆå§‹è§£ã€‚
    :param cities: åŸå¸‚çš„åæ ‡åˆ—è¡¨
    :return: æ„é€ å‡ºçš„åˆå§‹è§£è·¯å¾„
    """
    num_cities = len(cities)
    unvisited = list(range(num_cities))  # æœªè®¿é—®çš„åŸå¸‚åˆ—è¡¨
    solution = []
    current_city = random.choice(unvisited)  # éšæœºé€‰æ‹©ä¸€ä¸ªèµ·å§‹åŸå¸‚
    solution.append(current_city)
    unvisited.remove(current_city)

    # æ„é€ è·¯å¾„ï¼Œç›´åˆ°æ‰€æœ‰åŸå¸‚éƒ½è¢«è®¿é—®
    while unvisited:
        # æ„å»ºå€™é€‰åˆ—è¡¨ï¼Œå¹¶æŒ‰è·ç¦»æ’åº
        candidate_list = []
        for city in unvisited:
            distance = calculate_distance(cities[current_city], cities[city])
            candidate_list.append((city, distance))

        # æŒ‰è·ç¦»æ’åºï¼Œå¹¶æ„é€ å—é™å€™é€‰åˆ—è¡¨ (RCL)
        candidate_list.sort(key=lambda x: x[1])
        rcl_size = max(1, len(candidate_list) // 3)  # é€‰æ‹©å‰ 1/3 çš„è¾ƒä¼˜è§£ä½œä¸º RCL
        rcl = candidate_list[:rcl_size]

        # ä» RCL ä¸­éšæœºé€‰æ‹©ä¸€ä¸ªåŸå¸‚
        next_city = random.choice(rcl)[0]
        solution.append(next_city)
        unvisited.remove(next_city)
        current_city = next_city

    return solution

def local_search(solution, cities):
    """
    ä½¿ç”¨ 2-opt ç®—æ³•è¿›è¡Œå±€éƒ¨æœç´¢ä¼˜åŒ–ã€‚
    :param solution: åˆå§‹è§£è·¯å¾„
    :param cities: åŸå¸‚çš„åæ ‡åˆ—è¡¨
    :return: ä¼˜åŒ–åçš„è·¯å¾„
    """
    best_solution = solution[:]
    best_distance = total_distance(best_solution, cities)

    improved = True
    while improved:
        improved = False
        # 2-opt æœç´¢ï¼šäº¤æ¢è·¯å¾„ä¸­çš„ä¸¤ä¸ªè¾¹
        for i in range(1, len(solution) - 1):
            for j in range(i + 1, len(solution)):
                if j - i == 1:  # ç›¸é‚»åŸå¸‚è·³è¿‡
                    continue
                # äº¤æ¢ä¸¤ä¸ªåŸå¸‚ä¹‹é—´çš„è·¯å¾„
                new_solution = best_solution[:]
                new_solution[i:j] = best_solution[i:j][::-1]
                new_distance = total_distance(new_solution, cities)
                if new_distance < best_distance:
                    best_solution = new_solution[:]
                    best_distance = new_distance
                    improved = True

    return best_solution

def grasp_tsp(cities, max_iterations):
    """
    ä½¿ç”¨ GRASP ç®—æ³•æ±‚è§£æ—…è¡Œå•†é—®é¢˜ã€‚
    :param cities: åŸå¸‚çš„åæ ‡åˆ—è¡¨

    :return: æœ€ä¼˜è·¯å¾„å’Œæœ€çŸ­è·ç¦»

    best_solution = None

        initial_solution = greedy_construct_solution(cities)
        # å±€éƒ¨æœç´¢ä¼˜åŒ–è§£
        improved_solution = local_search(initial_solution, cities)

        current_distance = total_distance(improved_solution, cities)

        # æ›´æ–°å…¨å±€æœ€ä¼˜è§£
        if current_distance < best_distance:

    return best_solution, best_distance
