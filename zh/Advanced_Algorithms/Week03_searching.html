def treasure_searching(maze, blank):
    consistant_i = []
    count_i = []
    for i in range(len(maze)):
        if maze[i] == blank:
            consistant_i.append(i)

        else:
            if len(consistant_i) > 0:
                count_i.append(consistant_i.copy())
                consistant_i.clear()

    max_lst = count_i[0]
    for lst in count_i:
        if len(lst) > len(max_lst):
            max_lst = lst
    return max_lst
