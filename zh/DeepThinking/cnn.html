# --------------------- 1. å®šä¹‰ LeNet ç½‘ç»œ ---------------------
class LeNet(nn.Module):
    def __init__(self, use_maxpool=True):
        super(LeNet, self).__init__()

        # å·ç§¯å±‚
        self.conv1 = nn.Conv2d(1, 6, kernel_size=5, padding=2)  # 28x28 -> 28x28
        self.conv2 = nn.Conv2d(6, 16, kernel_size=5)  # 14x14 -> 10x10

        # é€‰æ‹©æ± åŒ–æ–¹å¼ï¼ˆé»˜è®¤ä½¿ç”¨ MaxPoolï¼‰
        self.pool = nn.MaxPool2d(kernel_size=2, stride=2) if use_maxpool else nn.AvgPool2d(kernel_size=2, stride=2)

        # å…¨è¿æ¥å±‚
        self.fc1 = nn.Linear(16 * 5 * 5, 120)
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, 10)

    def forward(self, x):
        x = self.pool(torch.relu(self.conv1(x)))  # ç¬¬ä¸€å±‚å·ç§¯ + ReLU + æ± åŒ–
        x = self.pool(torch.relu(self.conv2(x)))  # ç¬¬äºŒå±‚å·ç§¯ + ReLU + æ± åŒ–
        x = torch.flatten(x, 1)  # å±•å¹³
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        x = self.fc3(x)  # æœ€ç»ˆè¾“å‡º 10 ç»´
        return x

import torch
import torch.nn as nn
import torch.optim as optim
import torchvision
import torchvision.transforms as transforms

# --------------------- 1. å®šä¹‰ LeNet ç½‘ç»œ ---------------------
class LeNet(nn.Module):
    def __init__(self, use_maxpool=True, conv_channels=(6, 16), activation="relu", kernel_size=(5, 5), fc_sizes=(120, 84)):
        super(LeNet, self).__init__()

        # é€‰æ‹©æ¿€æ´»å‡½æ•°
        if activation == "relu":
            self.activation = nn.ReLU()
        elif activation == "sigmoid":
            self.activation = nn.Sigmoid()
        else:
            self.activation = nn.Tanh()

        # å·ç§¯å±‚
        self.conv1 = nn.Conv2d(1, conv_channels[0], kernel_size=kernel_size[0], padding=2)  # 28x28 -> 28x28
        self.conv2 = nn.Conv2d(conv_channels[0], conv_channels[1], kernel_size=kernel_size[1])  # 14x14 -> 10x10

        # é€‰æ‹©æ± åŒ–æ–¹å¼ï¼ˆé»˜è®¤ä½¿ç”¨ MaxPoolï¼‰
        self.pool = nn.MaxPool2d(kernel_size=2, stride=2) if use_maxpool else nn.AvgPool2d(kernel_size=2, stride=2)

        # è®¡ç®—å±•å¹³åçš„ç‰¹å¾å›¾å°ºå¯¸
        fc_input_size = conv_channels[1] * 5 * 5

        # å…¨è¿æ¥å±‚ï¼ˆå¯è°ƒæ•´å¤§å°ï¼‰
        self.fc1 = nn.Linear(fc_input_size, fc_sizes[0])
        self.fc2 = nn.Linear(fc_sizes[0], fc_sizes[1])
        self.fc3 = nn.Linear(fc_sizes[1], 10)

    def forward(self, x):
        x = self.pool(self.activation(self.conv1(x)))  # ç¬¬ä¸€å±‚å·ç§¯ + æ¿€æ´» + æ± åŒ–
        x = self.pool(self.activation(self.conv2(x)))  # ç¬¬äºŒå±‚å·ç§¯ + æ¿€æ´» + æ± åŒ–
        x = torch.flatten(x, 1)  # å±•å¹³
        x = self.activation(self.fc1(x))
        x = self.activation(self.fc2(x))
        x = self.fc3(x)  # æœ€ç»ˆè¾“å‡º 10 ç»´
        return x

# --------------------- 2. è®­ç»ƒæ¨¡å‹ ---------------------
def train_model(model, train_loader, test_loader, epochs=10, learning_rate=0.001):
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    model.to(device)

    criterion = nn.CrossEntropyLoss()
    optimizer = optim.Adam(model.parameters(), lr=learning_rate)

    for epoch in range(epochs):
        model.train()
        running_loss = 0.0
        for images, labels in train_loader:
            images, labels = images.to(device), labels.to(device)
            optimizer.zero_grad()
            outputs = model(images)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            running_loss += loss.item()

        print(f"Epoch {epoch + 1}, Loss: {running_loss / len(train_loader):.4f}")

    # æµ‹è¯•å‡†ç¡®ç‡
    model.eval()
    correct = 0
    total = 0
    with torch.no_grad():
        for images, labels in test_loader:
            images, labels = images.to(device), labels.to(device)
            outputs = model(images)
            _, predicted = torch.max(outputs, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()

    accuracy = 100 * correct / total
    print(f"Test Accuracy: {accuracy:.2f}%")
    return accuracy

import torch
import torch.nn as nn
import torchvision.transforms as transforms
import torchvision
import matplotlib.pyplot as plt
import numpy as np
from PIL import Image


# --------------------- 1. å®šä¹‰ LeNet ç½‘ç»œ ---------------------
class LeNet(nn.Module):
    def __init__(self):
        super(LeNet, self).__init__()
        self.conv1 = nn.Conv2d(1, 6, kernel_size=5, padding=2)
        self.pool = nn.MaxPool2d(kernel_size=2, stride=2)
        self.conv2 = nn.Conv2d(6, 16, kernel_size=5)

    def forward(self, x):
        x1 = torch.relu(self.conv1(x))  # ç¬¬ä¸€å±‚æ¿€æ´»å€¼
        x1_pooled = self.pool(x1)
        x2 = torch.relu(self.conv2(x1_pooled))  # ç¬¬äºŒå±‚æ¿€æ´»å€¼
        x2_pooled = self.pool(x2)
        return x1, x2


# --------------------- 2. è½½å…¥å¹¶é¢„å¤„ç†å›¾ç‰‡ ---------------------
def load_image(image_path):
    transform = transforms.Compose([
        transforms.Grayscale(),
        transforms.Resize((28, 28)),
        transforms.ToTensor(),
        transforms.Normalize((0.5,), (0.5,))
    ])
    image = Image.open(image_path)
    image = transform(image).unsqueeze(0)  # å¢åŠ  batch ç»´åº¦
    return image


# --------------------- 3. å¯è§†åŒ–æ¿€æ´»å€¼ ---------------------
def visualize_activations(activations, layer_name):
    activations = activations.detach().cpu().numpy()
    num_filters = activations.shape[1]
    fig, axes = plt.subplots(1, num_filters, figsize=(15, 5))
    for i in range(num_filters):
        axes[i].imshow(activations[0, i], cmap='gray')
        axes[i].axis('off')
    plt.suptitle(f'Activations of {layer_name}')
    plt.show()
