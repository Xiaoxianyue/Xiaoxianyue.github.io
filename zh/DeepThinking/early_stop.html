import torch
from torch import nn
import torchvision
import torchvision.transforms as transforms
import matplotlib.pyplot as plt

# è®¾å®šè¶…å‚æ•°
batch_size = 256
num_inputs, num_outputs = 784, 10
num_hiddens = 128
num_layers = 2
learning_rate = 0.1
num_epochs = 10
lambda_values = [0, 0.0001, 0.001, 0.01, 0.1, 1]  # ä¸åŒçš„æƒé‡è¡°å‡ç³»æ•° Î»

# é¢„å¤„ç† FashionMNIST æ•°æ®é›†
transform = transforms.Compose([transforms.ToTensor()])
train_dataset = torchvision.datasets.FashionMNIST(root="./data", train=True, transform=transform, download=True)
test_dataset = torchvision.datasets.FashionMNIST(root="./data", train=False, transform=transform, download=True)
train_loader = torch.utils.data.DataLoader(train_dataset, batch_size=batch_size, shuffle=True)
test_loader = torch.utils.data.DataLoader(test_dataset, batch_size=batch_size, shuffle=False)

# å®šä¹‰ MLP ç½‘ç»œ
class MLP(nn.Module):
    def __init__(self, num_inputs, num_hiddens, num_outputs, num_layers):
        super(MLP, self).__init__()
        layers = []
        layers.append(nn.Linear(num_inputs, num_hiddens))
        layers.append(nn.ReLU())

        for _ in range(num_layers - 1):
            layers.append(nn.Linear(num_hiddens, num_hiddens))
            layers.append(nn.ReLU())

        layers.append(nn.Linear(num_hiddens, num_outputs))
        self.net = nn.Sequential(*layers)

    def forward(self, X):
        return self.net(X.view(-1, num_inputs))  # å±•å¹³è¾“å…¥

# è®­ç»ƒå‡½æ•°ï¼Œæ¥æ”¶ä¸åŒçš„ weight_decay å‚æ•°
def train_with_weight_decay(lambda_values):
    train_losses, val_losses = [], []

    for weight_decay in lambda_values:
        net = MLP(num_inputs, num_hiddens, num_outputs, num_layers)
        loss = nn.CrossEntropyLoss()
        optimizer = torch.optim.SGD(net.parameters(), lr=learning_rate, weight_decay=weight_decay)  # æ·»åŠ æƒé‡è¡°å‡

        train_loss, val_loss = [], []
        for epoch in range(num_epochs):
            net.train()
            total_loss, total_samples = 0, 0
            for X, y in train_loader:
                y_hat = net(X)
                l = loss(y_hat, y)
                optimizer.zero_grad()
                l.backward()
                optimizer.step()
                total_loss += l.item() * y.size(0)
                total_samples += y.size(0)
            train_loss.append(total_loss / total_samples)

            net.eval()
            total, test_loss = 0, 0
            with torch.no_grad():
                for X, y in test_loader:
                    y_hat = net(X)
                    test_loss += loss(y_hat, y).item() * y.size(0)
                    total += y.size(0)
            val_loss.append(test_loss / total)

        train_losses.append(train_loss[-1])
        val_losses.append(val_loss[-1])

        print(f"Î»={weight_decay:.5f}, Final Train Loss={train_loss[-1]:.4f}, Final Val Loss={val_loss[-1]:.4f}")

    return train_losses, val_losses

# è®­ç»ƒå¹¶è®°å½•ä¸åŒ Î» å¯¹æŸå¤±çš„å½±å“
train_losses, val_losses = train_with_weight_decay(lambda_values)
