Task 02

创建Series数据对象，定义转换函数，然后用 map() 完成数据转换

代码

import pandas as pd
import numpy as np

# 创建英文成绩的 Series
english_score = pd.Series(
    np.random.randint(0, 100, size=10),
    name='英语'
)

# 定义成绩等级映射函数
def map_grade(x):
    if x >= 90:
        return 'A'
    elif x >= 80:
        return 'B'
    elif x >= 60:
        return 'C'
    else:
        return 'D'

# 创建 DataFrame 并添加等级列
df = pd.DataFrame({
    '英语': english_score,
    '等级': english_score.map(map_grade)
})

# 打印结果
print(df)
